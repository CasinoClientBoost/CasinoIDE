Обязательный импорт

Каждая программа на Casino++ должна начинаться с импорта библиотеки:

ИМПОРТИРОВАТЬ_CASINOCLIENT.CASINO++

Важно

Без этой строки в начале программы интерпретатор Casino++ откажется выполнять код. Это обязательное требование языка.

Тип Описание Пример

СТРОКА Текстовые данные в двойных кавычках "Джекпот!"
МАССИВ Коллекции элементов [1, 2, 3, "красное"]

Управляющие конструкции

Управление потоком выполнения программы реализовано с использованием казино-тематики.

Условные операторы

ПРОВЕРИТЬ_НА_НУРСУЛТАН (ставка > баланс) {
    ЦИТАТУ_НА_ЭКРАН "Недостаточно средств!";
}
ПРОВЕРИТЬ_НА_ДРУГОЙ_ЧИТ (ставка == 0) {
    ЦИТАТУ_НА_ЭКРАН "Ставка не может быть нулевой!";
}
ЕСЛИ_НЕТ_НУРСУЛТАНА {

    ЦИТАТУ_НА_ЭКРАН "Ставка принята!";
}

Примеры программ

Простая игра в рулетку

ИМПОРТИРОВАТЬ_CASINOCLIENT.CASINO++

ЧИСЛО баланс ПЖЖПЖПЖ_СТАВЬ_НА 1000;

ПОКА_ДЕПАЕТ (баланс > 0) {
    ЦИТАТУ_НА_ЭКРАН "Ваш баланс: " И_ЕЩЁ_НЕ_ЗАБУДЬ баланс;
    ЧИСЛО ставка ПЖЖПЖПЖ_СТАВЬ_НА ЗАПРОСИТЬ_ЦИТАТУ "Ваша ставка: ";
    
    ПРОВЕРИТЬ_НА_НУРСУЛТАН (ставка > баланс) {
        ЦИТАТУ_НА_ЭКРАН "Недостаточно средств!";
        ПРОДОЛЖИТЬ_ИГРУ;
    }
    
    ЧИСЛО результат ПЖЖПЖПЖ_СТАВЬ_НА ДЕПАЙ_РУЛЕТКУ(0, 36);
    ЦИТАТУ_НА_ЭКРАН "Выпало: " И_ЕЩЁ_НЕ_ЗАБУДЬ результат;
    
    ПРОВЕРИТЬ_НА_НУРСУЛТАН (результат == 7) {
        ЧИСЛО выигрыш ПЖЖПЖПЖ_СТАВЬ_НА ставка * 10;
        баланс ПЖЖПЖПЖ_СТАВЬ_НА баланс + выигрыш;
        ЦИТАТУ_НА_ЭКРАН "Джекпот! Вы выиграли: " И_ЕЩЁ_НЕ_ЗАБУДЬ выигрыш;
    }
    ЕСЛИ_НЕТ_НУРСУЛТАНА {
        баланс ПЖЖПЖПЖ_СТАВЬ_НА баланс - ставка;
        ЦИТАТУ_НА_ЭКРАН "Вы проиграли ставку!";
    }
}

ЦИТАТУ_НА_ЭКРАН "Игра окончена! Ваш финальный баланс: " И_ЕЩЁ_НЕ_ЗАБУДЬ баланс;

Угадай число

ИМПОРТИРОВАТЬ_CASINOCLIENT.CASINO++

ЧИСЛО загаданное_число ПЖЖПЖПЖ_СТАВЬ_НА ДЕПАЙ_РУЛЕТКУ(1, 100);
ЧИСЛО попытки ПЖЖПЖПЖ_СТАВЬ_НА 0;
ЧИСЛО максимальные_попытки ПЖЖПЖПЖ_СТАВЬ_НА 7;

ЦИТАТУ_НА_ЭКРАН "Угадайте число от 1 до 100! У вас " И_ЕЩЁ_НЕ_ЗАБУДЬ максимальные_попытки И_ЕЩЁ_НЕ_ЗАБУДЬ " попыток.";

ПОКА_ДЕПАЕТ (попытки < максимальные_попытки) {
    ЧИСЛО попытка ПЖЖПЖПЖ_СТАВЬ_НА ЗАПРОСИТЬ_ЦИТАТУ "Попытка " И_ЕЩЁ_НЕ_ЗАБУДЬ (попытки + 1) И_ЕЩЁ_НЕ_ЗАБУДЬ ": ";
    попытки ПЖЖПЖПЖ_СТАВЬ_НА попытки + 1;
    
    ПРОВЕРИТЬ_НА_НУРСУЛТАН (попытка == загаданное_число) {
        ЦИТАТУ_НА_ЭКРАН "Поздравляем! Вы угадали за " И_ЕЩЁ_НЕ_ЗАБУДЬ попытки И_ЕЩЁ_НЕ_ЗАБУДЬ " попыток!";
        ВЫЙТИ_ОТСЮДА;
    }
    ПРОВЕРИТЬ_НА_ДРУГОЙ_ЧИТ (попытка < загаданное_число) {
        ЦИТАТУ_НА_ЭКРАН "Слишком мало!";
    }
    ЕСЛИ_НЕТ_НУРСУЛТАНА {
        ЦИТАТУ_НА_ЭКРАН "Слишком много!";
    }
}

ЦИТАТУ_НА_ЭКРАН "Вы проиграли! Загаданное число было: " И_ЕЩЁ_НЕ_ЗАБУДЬ загаданное_число;
